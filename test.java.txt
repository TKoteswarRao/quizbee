package com.datastructure;

import java.util.HashMap;
import java.util.TreeMap;

public class Test {
	public static void main(String[] args) {
		HashMap hashMap = new HashMap();
		hashMap.put("a", 5);
		hashMap.put("d", 4);
		hashMap.put("b", 2);
		hashMap.put("a", 6);
		hashMap.put("c", 3);
		System.out.println("Test.main()" + hashMap);

		TreeMap map = new TreeMap();
		map.put("a", 5);
		map.put("d", 4);
		map.put("b", 2);
		map.put("a", 6);
		map.put("s", 0);
		map.put("x", 4);
		map.put("c", 3);
		System.out.println("Test.main()" + map);
		
	}
}
